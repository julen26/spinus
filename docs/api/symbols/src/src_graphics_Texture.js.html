 13 /**
 14 * Constructs Texture objects
 15 * @class Represents a Texture object
 16 * @param {sp.Context} context Context
 17 * @param {bool} [smooth=false] Color
 18 * @param {bool} [repeat=false] Color
 19 * @param {bool} [forcePOT=false] Color
 20 */
 21 sp.Texture = function(context, smooth, repeat, forcePOT) {
 22     /** @private */
 23 	this.context_ = context;
 24     /** @private */
 25     this.smooth_ = smooth || false;
 26     /** @private */
 27     this.repeat_ = repeat || false;
 28     /** @private */
 29     this.forcePOT_ = forcePOT || false;
 30 
 31     /** @private */
 32     this.image_ = null;
 33     /** @private */
 34     this.textureId_ = null;
 35 
 36     /** @private */
 37     this.size_ = new sp.Vector2();
 38 
 39     /** @private */
 40     this.callback_ = null;
 41 };
 42 
 43 /**
 44 * Loads the image from file
 45 *
 46 * @method
 47 * @param {string} sourcePath Path of the image file must be in the same domain
 48 * @param {function} callback Callback function
 49 */
 50 sp.Texture.prototype.loadFromFile = function (sourcePath, callback) {
 51     this.image_ = new Image();
 52 	this.callback_ = callback;
 53     this.image_.src = sourcePath;
 54     this.image_.onload = this.handleLoadedTexture.bind(this);
 55 };
 56 
 57 /**
 58 * Loads the image from HTML image element
 59 *
 60 * @method
 61 * @param {HTMLImageElement} image HTML image element
 62 */
 63 sp.Texture.prototype.loadFromImage = function (image) {
 64     this.image_ = image;
 65     this.handleLoadedTexture();
 66 };
 67 
 68 /** @private */
 69 sp.Texture.prototype.handleLoadedTexture = function () {
 70 		var gl = this.context_.GL();
 71 
 72         this.size_ = new sp.Vector2(this.image_.width, this.image_.height);
 73 
 74     	this.textureId_ = gl.createTexture();
 75     	gl.bindTexture(gl.TEXTURE_2D, this.textureId_);
 76 
 77         if (this.forcePOT_ && !this.isPowerOfTwo()) {
 78             var ctx = document.createElement('canvas').getContext('2d');
 79             ctx.canvas.width = this.nextHighestPowerOfTwo(this.size_.x);
 80             ctx.canvas.height = this.nextHighestPowerOfTwo(this.size_.y);
 81             ctx.drawImage(this.image_, 0, 0, this.image_.width, this.image_.height);
 82             this.size_.x = ctx.canvas.width;
 83             this.size_.y = ctx.canvas.height;
 84             this.image_ = ctx.canvas;
 85         }
 86 
 87     	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.image_);
 88         if (this.isPowerOfTwo()) {
 89             gl.generateMipmap(gl.TEXTURE_2D);
 90             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, (this.smooth_) ? gl.LINEAR_MIPMAP_NEAREST : gl.NEAREST);
 91             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, (this.repeat_) ? gl.REPEAT : gl.CLAMP_TO_EDGE);
 92             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, (this.repeat_) ? gl.REPEAT : gl.CLAMP_TO_EDGE);
 93         }
 94         else {
 95             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, (this.smooth_) ? gl.LINEAR : gl.NEAREST);
 96             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
 97             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
 98             this.repeat_ = false;
 99         }
100         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, (this.smooth_) ? gl.LINEAR : gl.NEAREST);
101 
102     	gl.bindTexture(gl.TEXTURE_2D, null);
103 
104         if (this.callback_) {
105     	   this.callback_();
106         }
107 };

129 /**
130 * Enable or disable smooth mode.
131 *
132 * @method
133 * @param {bool} smooth Enable smooth mode
134 */
135 sp.Texture.prototype.setSmooth = function (smooth) {
136     gl.bindTexture(gl.TEXTURE_2D, this.textureId_);
137     if (this.isPowerOfTwo()) {
138         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, (this.smooth_) ? gl.LINEAR_MIPMAP_NEAREST : gl.NEAREST);
139     }
140     else {
141         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, (this.smooth_) ? gl.LINEAR : gl.NEAREST);
142     }
143     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, (this.smooth_) ? gl.LINEAR : gl.NEAREST);
144     this.smooth_ = smooth;
145 };
146 
147 /**
148 * Enable or disable repeat mode.
149 *
150 * @method
151 * @param {bool} repeat Enable repeat mode
152 */
153 sp.Texture.prototype.setRepeat = function (repeat) {
154     gl.bindTexture(gl.TEXTURE_2D, this.textureId_);
155     if (this.isPowerOfTwo()) {
156         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, (repeat) ? gl.REPEAT : gl.CLAMP_TO_EDGE);
157         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, (repeat) ? gl.REPEAT : gl.CLAMP_TO_EDGE);
158         this.repeat_ = repeat;
159     }
160     else {
161         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
162         gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
163         this.repeat_ = false;
164     }
165 };

167 /**
168 * Checks wether or not texture size is power of two
169 *
170 * @method
171 * @returns {bool} True if texture size is power of two
172 */
173 sp.Texture.prototype.isPowerOfTwo = function () {
174     if (!this.size_) return false;
175     return ((this.size_.x & (this.size_.x - 1)) == 0) && ((this.size_.y & (this.size_.y - 1)) == 0);
176 };

179 /**
180 * Get next highest power of two
181 *
182 * @method
183 * @param {int} p Integer
184 * @returns {int} Next highest power of two
185 */
186 sp.Texture.prototype.nextHighestPowerOfTwo = function (p) {
187     p--;
188     for (var i = 1; i <= 16; i <<= 1) {
189         p |= p >> i;
190     }
191     return p + 1;
192 };
