  8 goog.provide('sp.Vertex');
  9 goog.require('sp');
 10 goog.require('sp.Vector2');
 11 goog.require('sp.Color');
 12 
 13 /**
 14 * Constructs Vertex objects
 15 * @class Represents a Vertex object
 16 * @param {sp.Vector2} position Position
 17 * @param {sp.Color} color Color
 18 * @param {sp.Vector2} texCoords Texture coordinates
 19 */
 20 sp.Vertex = function(position, color, texCoords) {
 21     /**
 22     * Position
 23     * @type sp.Vector2
 24     */
 25     this.position = position || new sp.Vector2(0, 0);
 26     /**
 27     * Color
 28     * @type sp.Color
 29     */
 30     this.color = color || new sp.Color(255, 255, 255);
 31     /**
 32     * Texture coordinates
 33     * @type sp.Vector2
 34     */
 35     this.texCoords = texCoords || new sp.Vector2(0, 0);
 36 };
 37 
 38 /**
 39 * Sets vertex components
 40 * @method
 41 * @param {sp.Vector2} position Position
 42 * @param {sp.Color} color Color
 43 * @param {sp.Vector2} texCoords Texture coordinates
 44 */
 45 sp.Vertex.prototype.set = function (position, color, texCoords) {
 46     this.position = position;
 47     this.color = color;
 48     this.texCoords = texCoords;
 49 };
 50 
 51 /**
 52 * Compares vertices and checks if they are equal
 53 * @method
 54 * @return {bool} True if vertices are equal
 55 */
 56 sp.Vertex.prototype.equals = function (vertex) {
 57     return (this.position.equals(vertex.position) && this.color.equals(vertex.color) && this.texCoords.equals(vertex.texCoords));
 58 };
