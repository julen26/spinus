  7 
  8 goog.provide('sp.Transform');
  9 goog.require('sp');
 10 goog.require('sp.Vector2');
 11 
 12 /**
 13 * Constructs Transform objects
 14 * @class Represents a 4x4 matrix
 15 */
 16 sp.Transform = function() {
 17     /** @private */
 18     this.matrix_ = [];
 19     
 20     this.matrix_[0] = 1.0; this.matrix_[4] = 0.0; this.matrix_[8]  = 0.0; this.matrix_[12] = 0.0;
 21     this.matrix_[1] = 0.0; this.matrix_[5] = 1.0; this.matrix_[9]  = 0.0; this.matrix_[13] = 0.0;
 22     this.matrix_[2] = 0.0; this.matrix_[6] = 0.0; this.matrix_[10] = 1.0; this.matrix_[14] = 0.0;
 23     this.matrix_[3] = 0.0; this.matrix_[7] = 0.0; this.matrix_[11] = 0.0; this.matrix_[15] = 1.0;
 24 };
 25 
 26 /**
 27 * Sets the matrix
 28 *
 29 * @method
 30 * @param {float} a00 0,0 component
 31 * @param {float} a01 0,1 component
 32 * @param {float} a02 0,2 component
 33 * @param {float} a10 1,0 component
 34 * @param {float} a11 1,1 component
 35 * @param {float} a12 1,2 component
 36 * @param {float} a20 2,0 component
 37 * @param {float} a21 2,1 component
 38 * @param {float} a22 2,2 component
 39 */
 40 sp.Transform.prototype.set = function (a00, a01, a02,
 41                                         a10, a11, a12,
 42                                         a20, a21, a22) {
 43     this.matrix_[0] = a00; this.matrix_[4] = a01; this.matrix_[8]  = 0.0; this.matrix_[12] = a02;
 44     this.matrix_[1] = a10; this.matrix_[5] = a11; this.matrix_[9]  = 0.0; this.matrix_[13] = a12;
 45     this.matrix_[2] = 0.0; this.matrix_[6] = 0.0; this.matrix_[10] = 1.0; this.matrix_[14] = 0.0;
 46     this.matrix_[3] = a20; this.matrix_[7] = a21; this.matrix_[11] = 0.0; this.matrix_[15] = a22;
 47 };
 48 
 49 /**
 50 * Get the internal matrix
 51 *
 52 * @method
 53 * @returns {float[]} 4x4 matrix
 54 */
 55 sp.Transform.prototype.getMatrix = function () {
 56     return this.matrix_;
 57 };
 58 
 59 /**
 60 * Gets new inverse transform
 61 *
 62 * @method
 63 * @returns {sp.Transform} Inverse transform or identity if determinant is zero
 64 */
 65 sp.Transform.prototype.getInverse = function () {
 66     var det =   this.matrix_[0] * (this.matrix_[15] * this.matrix_[5] - this.matrix_[7] * this.matrix_[13]) -
 67                 this.matrix_[1] * (this.matrix_[15] * this.matrix_[4] - this.matrix_[7] * this.matrix_[12]) +
 68                 this.matrix_[3] * (this.matrix_[13] * this.matrix_[4] - this.matrix_[5] * this.matrix_[12]);
 69     if (det != 0) {
 70         var inverse = new sp.Transform();
 71         inverse.set(  (this.matrix_[15] * this.matrix_[5] - this.matrix_[7] * this.matrix_[13]) / det,
 72                      -(this.matrix_[15] * this.matrix_[4] - this.matrix_[7] * this.matrix_[12]) / det,
 73                       (this.matrix_[13] * this.matrix_[4] - this.matrix_[5] * this.matrix_[12]) / det,
 74                      -(this.matrix_[15] * this.matrix_[1] - this.matrix_[3] * this.matrix_[13]) / det,
 75                       (this.matrix_[15] * this.matrix_[0] - this.matrix_[3] * this.matrix_[12]) / det,
 76                      -(this.matrix_[13] * this.matrix_[0] - this.matrix_[1] * this.matrix_[12]) / det,
 77                       (this.matrix_[7]  * this.matrix_[1] - this.matrix_[3] * this.matrix_[5])  / det,
 78                      -(this.matrix_[7]  * this.matrix_[0] - this.matrix_[3] * this.matrix_[4])  / det,
 79                       (this.matrix_[5]  * this.matrix_[0] - this.matrix_[1] * this.matrix_[4])  / det);
 80     }
 81     else {
 82         return new sp.Transform();
 83     }
 84 };

 86 /**
 87 * Combines the given transform
 88 *
 89 * @method
 90 * @param {sp.Transform} transform Transform to combine
 91 * @returns {sp.Transform} New combined transform
 92 */
 93 sp.Transform.prototype.combine = function (transform) {
 94     var a = this.matrix_;
 95     var b = transform.getMatrix();
 96     var a00 = a[0] * b[0] + a[4] * b[1] + a[12] * b[3];
 97     var a01 = a[0] * b[4] + a[4] * b[5] + a[12] * b[7];
 98     var a02 = a[0] * b[12] + a[4] * b[13] + a[12] * b[15];
 99     var a10 = a[1] * b[0] + a[5] * b[1] + a[13] * b[3];
100     var a11 = a[1] * b[4] + a[5] * b[5] + a[13] * b[7];
101     var a12 = a[1] * b[12] + a[5] * b[13] + a[13] * b[15];
102     var a20 = a[3] * b[0] + a[7] * b[1] + a[15] * b[3];
103     var a21 = a[3] * b[4] + a[7] * b[5] + a[15] * b[7];
104     var a22 = a[3] * b[12] + a[7] * b[13] + a[15] * b[15];
105     this.set(a00, a01, a02,
106              a10, a11, a12,
107              a20, a21, a22);
108     return this;
109 };

111 /**
112 * Translates the transform
113 *
114 * @method
115 * @param {float} x X translation
116 * @param {float} y Y translation
117 * @returns {sp.Transform} New translated transform
118 */
119 sp.Transform.prototype.translate = function (x, y) {
120     var translation = new sp.Transform();
121     translation.set(1, 0, x,
122                     0, 1, y,
123                     0, 0, 1);
124     return this.combine(translation);
125 };
126 
127 /**
128 * Scales the transform
129 *
130 * @method
131 * @param {float} x X scale factor
132 * @param {float} y Y scale factor
133 * @returns {sp.Transform} New scaled transform
134 */
135 sp.Transform.prototype.scale = function (x, y) {
136     var scaling = new sp.Transform();
137     scaling.set(x, 0, 0,
138                 0, y, 0,
139                 0, 0, 1);
140     return this.combine(scaling);
141 };
142 
143 /**
144 * Rotates the transform
145 *
146 * @method
147 * @param {float} angle Rotation angle in degrees
148 * @returns {sp.Transform} New rotated transform
149 */
150 sp.Transform.prototype.rotate = function (angle) {
151     var rad = angle * Math.PI / 180.0;
152     var cos = Math.cos(rad);
153     var sin = Math.sin(rad);
154 
155     var rotation = new sp.Transform();
156     rotation.set(cos,   -sin,   0,
157                  sin,    cos,   0,
158                  0,      0,     1);
159     return this.combine(rotation);
160 };
161 
162 /**
163 * Transforms a 2D point
164 *
165 * @method
166 * @param {float} x X coordinate
167 * @param {float} y Y coordinate
168 * @returns {sp.Vector2} New transformed point
169 */
170 sp.Transform.prototype.transformPoint = function (x, y) {
171     var vec = new Vector2();
172     vec.set(this.matrix_[0] * x + this.matrix_[4] * y + this.matrix_[12],
173             this.matrix_[1] * x + this.matrix_[5] * y + this.matrix_[13]);
174     return vec;
175 };
