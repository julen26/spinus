  8 goog.provide('sp.Rect');
  9 goog.require('sp');
 10 
 11 /**
 12 * Constructs Rect objects
 13 * @class Represents a Rect object
 14 * @param {float} x Left
 15 * @param {float} y Top
 16 * @param {float} w Width
 17 * @param {float} h Height
 18 */
 19 sp.Rect = function(x, y, w, h) {
 20     /**
 21     * X component
 22     * @type Float
 23     */
 24     this.x = x || 0;
 25     /**
 26     * Y component
 27     * @type Float
 28     */
 29     this.y = y || 0;
 30     /**
 31     * Width
 32     * @type Float
 33     */
 34     this.w = w || 0;
 35     /**
 36     * Height
 37     * @type Float
 38     */
 39     this.h = h || 0;
 40 };
 41 
 42 /**
 43 * Sets Rect properties
 44 *
 45 * @method
 46 * @param {float} x Left
 47 * @param {float} y Top
 48 * @param {float} w Width
 49 * @param {float} h Height
 50 */
 51 sp.Rect.prototype.set = function (x, y, w, h) {
 52     this.x = x;
 53     this.y = y;
 54     this.w = w;
 55     this.h = h;
 56 };
 57 
 58 /**
 59 * Compares Rect objects and checks if they are equal
 60 * 
 61 * @method
 62 * @param {sp.Rect} rect Rect to compare with
 63 * @return {bool} True if they are equal
 64 */
 65 sp.Rect.prototype.equals = function (rect) {
 66     return (this.x == rect.x && this.y == rect.y && this.w == rect.w && this.h == rect.h);
 67 };
