 11 /**
 12 * Constructs Color objects
 13 * @class Represents a Color object
 14 * @param {int} r Red component
 15 * @param {int} g Green component
 16 * @param {int} b Blue component
 17 * @param {int} a Alpha component
 18 */
 19 sp.Color = function(r, g, b, a) {
 20     /**
 21     * Red component
 22     * @type int
 23     */
 24     this.r = r || 0;
 25     /**
 26     * Green component
 27     * @type int
 28     */
 29     this.g = g || 0;
 30     /**
 31     * Blue component
 32     * @type int
 33     */
 34     this.b = b || 0;
 35     /**
 36     * Alpha component
 37     * @type int
 38     */
 39     this.a = a || 255;
 40 };
 41 
 42 /**
 43 * Compares colors and checks if they are equal
 44 *
 45 * @method
 46 * @return {bool} True if colors are equal
 47 */
 48 sp.Color.prototype.equals = function (color) {
 49     return (this.r == color.r && this.g == color.g && this.b == color.b && this.a == color.a);
 50 };

 52 /**
 53 * Gets a rgba string that represents the color
 54 *
 55 * @method
 56 * @return {string} rgba string
 57 */
 58 sp.Color.prototype.getRGBAString = function () {
 59     return 'rgba(' + Math.floor(this.r) + ',' + Math.floor(this.g) + ',' + Math.floor(this.b) + ',' + (this.a / 255) +')';
 60 };
 61 
 62 /**
 63 * Gets rgb color in hexadecimal format
 64 *
 65 * @method
 66 * @return {string} RGB in hex format
 67 */
 68 sp.Color.prototype.getHexString = function () {
 69     var r = Math.floor(this.r).toString(16);
 70     r = (r.length == 1) ? '0' + r : r;
 71     var g = Math.floor(this.g).toString(16);
 72     g = (g.length == 1) ? '0' + g : g;
 73     var b = Math.floor(this.b).toString(16);
 74     b = (b.length == 1) ? '0' + b : b;
 75     return '#' + r + g + b;
 76 };

 78 /**
 79 * Creates a new Color object from normalized color components (in range 0.0-1.0)
 80 *
 81 * @method
 82 * @param {float} r Red component
 83 * @param {float} g Green component
 84 * @param {float} b Blue component
 85 * @param {float} a Alpha component
 86 * @return {sp.Color} New color object
 87 */
 88 sp.Color.fromNormalized = function (r, g, b, a) {
 89     return new sp.Color(Math.floor(r * 255), Math.floor(g * 255), Math.floor(b * 255), Math.floor(a * 255));
 90 };
 91 
 92 /**
 93 * Creates a new Color object randomly
 94 *
 95 * @method
 96 * @return {sp.Color} New color object
 97 */
 98 sp.Color.random = function () {
 99     return sp.Color.fromNormalized(Math.random(), Math.random(), Math.random(), Math.random());
100 };
101 
102 /**
103 * Creates a new Color from a hexadecimal format
104 *
105 * @method
106 * @param {string} hex Color in hex format
107 * @return {sp.Color} New color object
108 */
109 sp.Color.fromHex = function (hex) {
110     var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
111     return result ? new sp.Color(parseInt(result[1], 16), parseInt(result[2], 16), parseInt(result[3], 16)) : new sp.Color();
112 };
