  8 goog.provide('sp.RectangleShape');
  9 goog.require('sp');
 10 goog.require('sp.extend');
 11 goog.require('sp.Shape');
 12 goog.require('sp.Vector2');
 13 goog.require('sp.Color');
 14 
 15 /**
 16 * Constructs RectangleShape objects
 17 * @class Represents a RectangleShape object
 18 * @extends sp.Shape
 19 * @param {float} w Width
 20 * @param {float} h Height
 21 * @param {sp.Color} color Color
 22 */
 23 sp.RectangleShape = function(w, h, color) {
 24     sp.Shape.call(this, 4);
 25 
 26     /** @private */
 27     this.size_ = null;
 28 
 29     this.setSize(w, h);
 30     this.setColor(color);
 31 };
 32 sp.extend(sp.RectangleShape, sp.Shape);
 33 
 34 /**
 35 * Sets rectangle size
 36 *
 37 * @method
 38 * @param {float} w Width
 39 * @param {float} h Height
 40 */
 41 sp.RectangleShape.prototype.setSize = function (w, h) {
 42     this.size_ = new sp.Vector2(w, h);
 43 
 44     this.setPointPosition(0, new sp.Vector2(0, 0));
 45     this.setPointPosition(1, new sp.Vector2(w, 0));
 46     this.setPointPosition(2, new sp.Vector2(w, h));
 47     this.setPointPosition(3, new sp.Vector2(0, h));
 48 };
 49 
 50 /**
 51 * Sets an uniform fill color
 52 *
 53 * @method
 54 * @param {sp.Color} color Fill color
 55 */
 56 sp.RectangleShape.prototype.setColor = function (color) {
 57     color = color || new sp.Color();
 58     
 59     this.setPointColor(0, color);
 60     this.setPointColor(1, color);
 61     this.setPointColor(2, color);
 62     this.setPointColor(3, color);
 63 };
 64 
 65 /**
 66 * Gets rectangle size
 67 *
 68 * @method
 69 * @returns {sp.Vector2} Rectangle size
 70 */
 71 sp.RectangleShape.prototype.getSize = function () {
 72     return this.size_;
 73 };
 74 
 75 /**
 76 * Sets a color for each vertex
 77 *
 78 * @method
 79 * @param {sp.Color} color1 Top-left color
 80 * @param {sp.Color} color2 Top-right color
 81 * @param {sp.Color} color3 Bottom-right color
 82 * @param {sp.Color} color4 Bottom-left color
 83 */
 84 sp.RectangleShape.prototype.setColors = function (color1, color2, color3, color4) {
 85     color1 = color1 || new sp.Color();
 86     color2 = color2 || new sp.Color();
 87     color3 = color3 || new sp.Color();
 88     color4 = color4 || new sp.Color();
 89 
 90     this.setPointColor(0, color1);
 91     this.setPointColor(1, color2);
 92     this.setPointColor(2, color3);
 93     this.setPointColor(3, color4);
 94 };
