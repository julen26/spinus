 19 /**
 20 * Constructs Sprite objects
 21 * @class Represents a Sprite object
 22 * @extends sp.Transformable
 23 * @extends sp.Drawable
 24 * @param {sp.Texture} texture Texture
 25 */
 26 sp.Sprite = function(texture) {
 27 	//Call base constructor
 28     sp.Transformable.call(this);
 29 
 30     /** @private */
 31 	this.vertexArray_ = new sp.VertexArray(sp.PrimitiveType.TRIANGLE_FAN, 4);
 32 
 33 	/** @private */
 34 	this.texture_ = null;
 35 	this.setTexture(texture);
 36 };
 37 sp.extend(sp.Sprite, sp.Transformable);
 38 sp.extend(sp.Sprite, sp.Drawable);
 39 
 40 /**
 41 * Set a new texture.
 42 *
 43 * @method
 44 * @param {sp.Texture} texture Texture
 45 */
 46 sp.Sprite.prototype.setTexture = function (texture) {
 47 	this.texture_ = texture;
 48 
 49 	if (this.texture_) {
 50 		var size = this.texture_.getSize();
 51 
 52 		this.vertexArray_.getVertex(0).position = new sp.Vector2(0, 0);
 53 		this.vertexArray_.getVertex(1).position = new sp.Vector2(size.x, 0);
 54 		this.vertexArray_.getVertex(2).position = new sp.Vector2(size.x, size.y);
 55 		this.vertexArray_.getVertex(3).position = new sp.Vector2(0, size.y);
 56 	}
 57 
 58 	this.vertexArray_.getVertex(0).texCoords = new sp.Vector2(0, 0);
 59     this.vertexArray_.getVertex(1).texCoords = new sp.Vector2(1, 0);
 60     this.vertexArray_.getVertex(2).texCoords = new sp.Vector2(1, 1);
 61     this.vertexArray_.getVertex(3).texCoords = new sp.Vector2(0, 1);
 62 
 63     this.textureRect_ = new sp.Rect(0, 0, 1, 1);
 64 };
 65 
 66 /**
 67 * Get the texture of the Sprite
 68 *
 69 * @method
 70 * @returns {sp.Texture} Texture
 71 */
 72 sp.Sprite.prototype.getTexture = function () {
 73 	return this.texture_;
 74 };
 75 
 76 /**
 77 * Set texture coordinates
 78 *
 79 * @method
 80 * @param {sp.Rect} rect Texture coordinates
 81 */
 82 sp.Sprite.prototype.setTextureRect = function (rect) {
 83 	if (this.texture_) {
 84 		var size = this.texture_.getSize();
 85 
 86 		this.vertexArray_.getVertex(0).position = new sp.Vector2(0, 0);
 87 		this.vertexArray_.getVertex(1).position = new sp.Vector2(size.x * rect.w, 0);
 88 		this.vertexArray_.getVertex(2).position = new sp.Vector2(size.x * rect.w, size.y * rect.h);
 89 		this.vertexArray_.getVertex(3).position = new sp.Vector2(0, size.y * rect.h);
 90 	}
 91 
 92 	this.textureRect_ = rect;
 93 	this.vertexArray_.getVertex(0).texCoords = new sp.Vector2(rect.x, rect.y);
 94     this.vertexArray_.getVertex(1).texCoords = new sp.Vector2(rect.x + rect.w, rect.y);
 95     this.vertexArray_.getVertex(2).texCoords = new sp.Vector2(rect.x + rect.w, rect.y + rect.h);
 96     this.vertexArray_.getVertex(3).texCoords = new sp.Vector2(rect.x, rect.y + rect.h);
 97 };
 98 
 99 /**
100 * Get the texture coordinates.
101 *
102 * @method
103 * @returns {sp.Rect} Texture coordinates
104 */
105 sp.Sprite.prototype.getTextureRect = function () {
106 	return this.textureRect_;
107 };
108 
109 /**
110 * Set sprite color.
111 *
112 * @method
113 * @param {sp.Color} color Color
114 */
115 sp.Sprite.prototype.setColor = function (color) {
116     color = color || new sp.Color();
117     
118     this.vertexArray_.getVertex(0).color = color;
119     this.vertexArray_.getVertex(1).color = color;
120     this.vertexArray_.getVertex(2).color = color;
121     this.vertexArray_.getVertex(3).color = color;
122 };
123 
124 /**
125 * Draws the sprite in the given context with optional render options.
126 *
127 * @method
128 * @param {sp.Context} context Context
129 * @param {sp.RenderOptions} renderOptions Optional render options
130 */
131 sp.Sprite.prototype.draw = function (context, renderOptions) {
132     renderOptions.transform = this.getTransform();
133     renderOptions.texture = this.texture_;
134     
135     this.vertexArray_.draw(context, renderOptions);
136 };
