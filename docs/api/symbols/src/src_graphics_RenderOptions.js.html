  8 goog.provide('sp.RenderOptions');
  9 goog.require('sp');
 10 goog.require('sp.BlendMode');
 11 goog.require('sp.Transform');
 12 
 13 /**
 14 * Constructs RenderOptions objects
 15 * @class Represents a RenderOptions object
 16 */
 17 sp.RenderOptions = function(transform, texture, shader, blendMode) {
 18     /**
 19     * Blend mode.
 20     * @type sp.BlendMode
 21     */
 22     this.blendMode = blendMode || sp.BlendMode.ALPHA;
 23     /**
 24     * Transform matrix.
 25     * @type sp.Transform
 26     */
 27     this.transform = transform || new sp.Transform();
 28     /**
 29     * Texture.
 30     * @type sp.Texture
 31     */
 32     this.texture = texture;
 33     /**
 34     * Shader.
 35     * @type sp.Shader
 36     */
 37     this.shader = shader;
 38 };
