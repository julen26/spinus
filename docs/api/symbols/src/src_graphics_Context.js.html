 16 /**
 17 * Constructs Context objects
 18 * @class Represents a Context object
 19 * @param {string} canvasId ID of the canvas element
 20 */
 21 sp.Context = function(canvasId) {
 22     /** @protected */
 23     this.canvas_ = document.getElementById(canvasId);
 24 
 25     /** @protected */
 26     this.gl_ = null;
 27 
 28     if (!this.canvas_) {
 29         throw 'Specified canvas element is missing.';
 30     }
 31 
 32     if (window.WebGLRenderingContext) {
 33         this.gl_ = this.canvas_.getContext('webgl') || this.canvas_.getContext('experimental-webgl');   
 34     }
 35     else {
 36         throw 'WebGL context is required and it\'s not supported by the browser.'
 37     }
 38 
 39     if (!this.gl_) {
 40         throw 'Unable to initialize a valid context. Your browser may not support it.'
 41     }
 42 
 43     /** @protected */
 44     this.viewportWidth_ = this.canvas_.width;
 45     /** @protected */
 46     this.viewportHeight_ = this.canvas_.height;
 47 
 48     /** @protected */
 49     this.defaultView_ = new sp.View(this.viewportWidth_, this.viewportHeight_);
 50     /** @protected */
 51     this.currentView_ = this.defaultView_;
 52 
 53     /** @protected */
 54     this.defaultShader_ = new sp.DefaultShader(this);
 55     /** @protected */
 56     this.defaultShaderTextured_ = new sp.DefaultShader(this, true);
 57 
 58     //Initialize buffer
 59     /** @protected */
 60     this.vertexPositionBuffer = this.gl_.createBuffer();
 61     /** @protected */
 62     this.vertexColorBuffer = this.gl_.createBuffer();
 63     /** @protected */
 64     this.vertexTexCoordsBuffer = this.gl_.createBuffer();
 65 
 66     this.gl_.viewport(0, 0, this.viewportWidth_, this.viewportHeight_);
 67     this.gl_.clearColor(0.0, 0.0, 0.0, 1.0);
 68     this.gl_.enable(this.gl_.BLEND);
 69     this.gl_.clear(this.gl_.COLOR_BUFFER_BIT); 
 70 };

152 /**
153 * Clear the context with an optional color
154 *
155 * @method
156 * @param {sp.Color} color Clear color
157 */
158 sp.Context.prototype.clear = function(color) {
159     color = color || new sp.Color();
160     this.gl_.clearColor(color.r / 255, color.g / 255, color.b / 255, color.a / 255);
161     this.gl_.clear(this.gl_.COLOR_BUFFER_BIT); 
162 }

164 /**
165 * Draws a drawable object
166 *
167 * @method
168 * @param {sp.Drawable} drawable Drawable object
169 * @param {sp.RenderOptions} renderOptions Optional render options
170 */
171 sp.Context.prototype.draw = function(drawable, renderOptions) {
172     drawable.draw(this, renderOptions || new sp.RenderOptions());
173 };

175 /**
176 * Draws vertices
177 *
178 * @method
179 * @param {List} vertices Vertices list
180 * @param {sp.PrimitiveType} type Primitive type
181 * @param {sp.RenderOptions} renderOptions Optional render options
182 */
183 sp.Context.prototype.drawVertices = function(vertices, type, renderOptions) {
184     var gl = this.GL();
185 
186     //Load default render options
187     if (!renderOptions) {
188         renderOptions = new sp.RenderOptions();
189         renderOptions.shader = this.getDefaultShader();
190     }
191 
192     //Set blend mode
193     var blendMode = renderOptions.blendMode;
194     if (!blendMode) {
195         blendMode = sp.BlendMode.ALPHA;
196     }
197     var factors = [gl.ZERO, gl.ONE, 
198         gl.SRC_COLOR, gl.ONE_MINUS_SRC_COLOR, gl.DST_COLOR, gl.ONE_MINUS_DST_COLOR,
199         gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA, gl.DST_ALPHA, gl.ONE_MINUS_DST_ALPHA,
200         gl.SRC_ALPHA_SATURATE];
201     var equations = [gl.FUNC_ADD, gl.FUNC_SUBTRACT, gl.FUNC_REVERSE_SUBTRACT];
202     this.gl_.blendFuncSeparate(factors[blendMode.srcColorFactor], factors[blendMode.dstColorFactor],
203                                 factors[blendMode.srcAlphaFactor], factors[blendMode.dstAlphaFactor]);
204     this.gl_.blendEquationSeparate(equations[blendMode.colorEquation], equations[blendMode.alphaEquation]);
205 
206     //Set shader
207     if (!renderOptions.shader) {
208         if (renderOptions.texture) {
209             renderOptions.shader = this.getDefaultShaderTextured();
210         }
211         else {
212             renderOptions.shader = this.getDefaultShader();
213         }
214     }
215     renderOptions.shader.use();
216 
217     //Positions
218     gl.bindBuffer(gl.ARRAY_BUFFER, this.vertexPositionBuffer);
219     var positions = sp.VertexArray.getPositionArray(vertices);
220     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(positions), gl.STATIC_DRAW);
221 
222     //Colors
223     gl.bindBuffer(gl.ARRAY_BUFFER, this.vertexColorBuffer);
224     var colors = sp.VertexArray.getColorArray(vertices);
225     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(colors), gl.STATIC_DRAW);
226 
227     //Texture coordinates
228     gl.bindBuffer(gl.ARRAY_BUFFER, this.vertexTexCoordsBuffer);
229     var texCoords = sp.VertexArray.getTexCoordsArray(vertices);
230     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(texCoords), gl.STATIC_DRAW);
231 
232     //Bind buffer to attributes
233     gl.bindBuffer(gl.ARRAY_BUFFER, this.vertexPositionBuffer);
234     var positionAttribute = renderOptions.shader.getAttribLocation('a_position');
235     gl.enableVertexAttribArray(positionAttribute);
236     gl.vertexAttribPointer(positionAttribute, 2, gl.FLOAT, false, 0, 0);
237 
238     gl.bindBuffer(gl.ARRAY_BUFFER, this.vertexColorBuffer);
239     var colorAttribute = renderOptions.shader.getAttribLocation('a_color');
240     gl.enableVertexAttribArray(colorAttribute);
241     gl.vertexAttribPointer(colorAttribute, 4, gl.FLOAT, false, 0, 0);
242 
243     if (renderOptions.texture) {
244         gl.bindBuffer(gl.ARRAY_BUFFER, this.vertexTexCoordsBuffer);
245         var texCoordsAttribute = renderOptions.shader.getAttribLocation('a_texCoord');
246         gl.enableVertexAttribArray(texCoordsAttribute);
247         gl.vertexAttribPointer(texCoordsAttribute, 2, gl.FLOAT, false, 0, 0);
248     }
249 
250     renderOptions.shader.uniformfv('u_resolution', [this.viewportWidth_, this.viewportHeight_]);
251     var transformMatrix = renderOptions.transform.getMatrix();
252     renderOptions.shader.uniformMatrixfv('u_transform', transformMatrix);
253     var viewMatrix = this.currentView_.getTransform().getMatrix();
254     renderOptions.shader.uniformMatrixfv('u_view', viewMatrix);
255     var projectionMatrix = this.currentView_.getProjection().getMatrix();
256     renderOptions.shader.uniformMatrixfv('u_projection', projectionMatrix);
257 
258     var viewport = this.currentView_.getViewport();
259     gl.viewport(viewport.x * this.viewportWidth_, this.viewportHeight_ - (viewport.y + viewport.h) * this.viewportHeight_, viewport.w * this.viewportWidth_, viewport.h * this.viewportHeight_);
260 
261     if (renderOptions.texture) {
262         gl.activeTexture(gl.TEXTURE0);
263         gl.bindTexture(gl.TEXTURE_2D, renderOptions.texture.getTextureId());
264     }
265 
266     //Find WebGL primitive type
267     var modes = [gl.POINTS, gl.LINES, gl.LINE_STRIP, gl.LINE_LOOP, gl.TRIANGLES, gl.TRIANGLE_STRIP, gl.TRIANGLE_FAN];
268     var mode = modes[type];
269 
270     //Draw primitives
271     gl.drawArrays(mode, 0, vertices.length);
272 
273     //TODO: Reset states
274     gl.bindTexture(gl.TEXTURE_2D, null);
275 };
