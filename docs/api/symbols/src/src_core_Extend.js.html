 11 /*
 12 * Extends the prototype of derived objects with the prototype of base object.
 13 * The contructor of the derived object must call the constructor of base object: 'base.call(this)'
 14 *
 15 * @method
 16 * @param {Object} derived Derived object
 17 * @param {Object} base Base object
 18 * @returns {Object} Returns the new prototype of the derived object
 19 */
 20 sp.extend = function (derived, base) {
 21     for(var key in base.prototype) {
 22         if(base.prototype.hasOwnProperty(key)) {
 23             derived.prototype[key] = base.prototype[key];
 24         }
 25     }
 26     return derived.prototype;
 27 }
