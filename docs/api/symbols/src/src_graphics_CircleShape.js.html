  8 goog.provide('sp.CircleShape');
  9 goog.require('sp');
 10 goog.require('sp.extend');
 11 goog.require('sp.Shape');
 12 
 13 /**
 14 * Constructs CircleShape objects
 15 * @class Represents a CircleShape object
 16 * @extends sp.Shape
 17 * @param {float} radius Radius
 18 * @param {sp.Color} color Color
 19 */
 20 sp.CircleShape = function(radius, color) {
 21     sp.Shape.call(this, 30);
 22 
 23     /** @private */
 24     this.radius_ = radius || 0;
 25 
 26     this.update();
 27 
 28     this.setColor(color);
 29 };
 30 sp.extend(sp.CircleShape, sp.Shape);
 31 
 32 /**
 33 * Sets radius of the circle
 34 *
 35 * @method
 36 * @param {float} radius Radius
 37 */
 38 sp.CircleShape.prototype.setRadius = function (radius) {
 39     this.radius_ = radius;
 40     this.update();
 41 };
 42 
 43 /**
 44 * Resizes the point list, adding default points or removing existing ones to match the new length. Increase this value to increase the quality but lose efficiency.
 45 * Default value is 30.
 46 *
 47 * @method
 48 * @param {int} pointCount Points
 49 */
 50 sp.CircleShape.prototype.setPointCount = function (pointCount) {
 51     sp.Shape.prototype.setPointCount.call(this, pointCount);
 52     this.update();
 53 };
 54 
 55 /**
 56 * Recalculates all the points again. Should be called if points are changed.
 57 *
 58 * @method
 59 */
 60 sp.CircleShape.prototype.update = function () {
 61     var pointCount = this.getPointCount();
 62     if (pointCount > 2) {
 63         var alfa = 2 * Math.PI / pointCount;
 64         for (var i = 0; i < pointCount; i++) {
 65             var angle = alfa * i;
 66             var pos = new sp.Vector2(Math.cos(angle) * this.radius_, Math.sin(angle) * this.radius_);
 67             this.setPointPosition(i, pos);
 68         }
 69     }
 70 };
 71 
 72 /**
 73 * Sets an uniform fill color
 74 *
 75 * @method
 76 * @param {sp.Color} color Fill color
 77 */
 78 sp.CircleShape.prototype.setColor = function (color) {
 79     color = color || new sp.Color();
 80     var pointCount = this.getPointCount();
 81     for (var i = 0; i < pointCount; i++) {
 82         this.setPointColor(i, color);
 83     }
 84 };
