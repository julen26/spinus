 19 /** 
 20 * Enum for text style
 21 * @enum {number}
 22 */
 23 sp.TextStyle = {
 24     NORMAL : 0,
 25     BOLD : 1,
 26     ITALIC : 2,
 27     UNDERLINE : 4,
 28     STRIKETHROUGH : 8
 29 };

 31 /**
 32 * Constructs Text objects
 33 * @class Represents a Text object
 34 * @extends sp.Transformable
 35 * @extends sp.Drawable
 36 * @param {string} [str=''] Text's string
 37 * @param {float} [width=100] Width
 38 * @param {float} [height=100] Height
 39 * @param {string} [font='Arial'] Font
 40 * @param {float} [characterSize=12] Character size
 41 * @param {sp.TextStyle} [style=sp.TextStyle.Normal] Text style
 42 * @param {string} [color=Black] Color
 43 */
 44 sp.Text = function(str, width, height, font, characterSize, style, color) {
 45     //Call base constructor
 46     sp.Transformable.call(this);
 47 
 48     /** @private */
 49     this.texture_ = null;
 50 
 51     /** @private */
 52     this.string_ = '';
 53 
 54     /** @private */
 55     this.width_ = width || 100;
 56     /** @private */
 57     this.height_ = height || 100;
 58 
 59     /** @private */
 60     this.font_ = font || 'Arial';
 61     /** @private */
 62     this.characterSize_ = characterSize || 12;
 63     /** @private */
 64     this.style_ = style || sp.TextStyle.NORMAL;
 65     /** @private */
 66     this.color_ = color || new sp.Color();
 67 
 68     /** @private */
 69     this.vertexArray_ = new sp.VertexArray(sp.PrimitiveType.TRIANGLE_FAN, 4);
 70     this.vertexArray_.getVertex(0).texCoords = new sp.Vector2(0, 0);
 71     this.vertexArray_.getVertex(1).texCoords = new sp.Vector2(1, 0);
 72     this.vertexArray_.getVertex(2).texCoords = new sp.Vector2(1, 1);
 73     this.vertexArray_.getVertex(3).texCoords = new sp.Vector2(0, 1);
 74     
 75     this.vertexArray_.getVertex(0).color = this.color_;
 76     this.vertexArray_.getVertex(1).color = this.color_;
 77     this.vertexArray_.getVertex(2).color = this.color_;
 78     this.vertexArray_.getVertex(3).color = this.color_;
 79 
 80     /** @private */
 81     this.needsTextureUpdate_ = true;
 82 
 83     this.setString(str);
 84 };
 85 sp.extend(sp.Text, sp.Transformable);
 86 sp.extend(sp.Text, sp.Drawable);

 88 /**
 89 * Set text content.
 90 *
 91 * @method
 92 * @param {string} str Text content
 93 */
 94 sp.Text.prototype.setString = function (str) {
 95     str = str || '';
 96     if (str != this.string_) {
 97         this.string_ = str;
 98         this.needsTextureUpdate_ = true;
 99     }
100 };

150 /**
151 * Updates the texture that contains the text.
152 *
153 * @method
154 * @param {sp.Context} context Context
155 */
156 sp.Text.prototype.updateTexture = function (context) {
157     var gl = context.GL();
158     var ctx = document.createElement('canvas').getContext('2d');
159 
160     if (ctx) {
161         ctx.width = this.width_;
162         ctx.height = this.height_;
163         ctx.canvas.width = this.width_;
164         ctx.canvas.height = this.height_;
165         ctx.fillStyle = '#ffffff';
166 
167         var font = this.characterSize_ + 'px ' + this.font_;
168         if ((this.style_ & sp.TextStyle.BOLD) != 0) {
169             font = 'bold ' + font;
170         }
171         if ((this.style_ & sp.TextStyle.ITALIC) != 0) {
172             font = 'italic ' + font;
173         }
174         ctx.font = font;
175 
176         ctx.textBaseline = 'top';
177         ctx.fillText(this.string_, 0, 0);
178 
179         //Underline and strikethrough
180         var textWidth = ctx.measureText(this.string_).width;
181         var lineHeight = this.characterSize_ / 15;
182         var startX;
183         var endX;
184         if (ctx.textAlign == 'center') {
185             startX = -textWidth / 2;
186             endX = textWidth / 2;
187         }
188         else if (ctx.textAlign == 'right') {
189             startX = -textWidth;
190             endX = 0;
191         }
192         else {
193             startX = 0;
194             endX = textWidth;
195         }
196         var lineY;
197         if ((this.style_ & sp.TextStyle.UNDERLINE) != 0) {
198             lineY = this.characterSize_;
199             ctx.beginPath();
200             ctx.strokeStyle = ctx.fillStyle;
201             ctx.lineWidth = lineHeight;
202             ctx.moveTo(startX, lineY);
203             ctx.lineTo(endX, lineY);
204             ctx.stroke();
205         }
206         if ((this.style_ & sp.TextStyle.STRIKETHROUGH) != 0) {
207             lineY = this.characterSize_ / 1.5;
208             ctx.beginPath();
209             ctx.strokeStyle = ctx.fillStyle;
210             ctx.lineWidth = lineHeight;
211             ctx.moveTo(startX, lineY);
212             ctx.lineTo(endX, lineY);
213             ctx.stroke();
214         }
215 
216         if (this.texture_) {
217             gl.deleteTexture(this.texture_.getTextureId());
218         }
219         this.texture_ = new sp.Texture(context);
220         this.texture_.loadFromImage(ctx.canvas);
221 
222         var size = this.texture_.getSize();
223         this.vertexArray_.getVertex(0).position = new sp.Vector2(0, 0);
224         this.vertexArray_.getVertex(1).position = new sp.Vector2(size.x, 0);
225         this.vertexArray_.getVertex(2).position = new sp.Vector2(size.x, size.y);
226         this.vertexArray_.getVertex(3).position = new sp.Vector2(0, size.y);
227 
228         this.needsTextureUpdate_ = false;
229     }
230 };
231 
232 /**
233 * Draws the text to the context. First updates the texture if needed.
234 *
235 * @method
236 * @param {sp.Context} context Context
237 * @param {sp.RenderOptions} renderOptions Optional render options
238 */
239 sp.Text.prototype.draw = function (context, renderOptions) {
240     if (this.needsTextureUpdate_) {
241         this.updateTexture(context);
242     }
243 
244     if (this.texture_) {
245         renderOptions.transform = this.getTransform();
246         renderOptions.texture = this.texture_;
247     
248         this.vertexArray_.draw(context, renderOptions);
249     }
250 };
