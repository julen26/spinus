 11 /**
 12 * Constructs Vector2 objects
 13 * @class Represents a Vector2 object
 14 * @param {float} x X component
 15 * @param {float} y Y component
 16 */
 17 sp.Vector2 = function(x, y) {
 18     /**
 19     * X component
 20     * @type Float
 21     */
 22     this.x = x || 0;
 23     /**
 24     * Y component
 25     * @type Float
 26     */
 27     this.y = y || 0;
 28 };
 29 
 30 /**
 31 * Sets vector components
 32 
 33 * @method
 34 */
 35 sp.Vector2.prototype.set = function (x, y) {
 36     this.x = x;
 37     this.y = y;
 38 };
 39 
 40 /**
 41 * Compares vectors and checks if they are equal
 42 *
 43 * @method
 44 * @param {sp.Vector2} vector Vector to compare with
 45 * @return {bool} True if vectors are equal
 46 */
 47 sp.Vector2.prototype.equals = function (vector) {
 48     return (this.x == vector.x && this.y == vector.y);
 49 };
 50 
 51 /**
 52 * Compute the normal of a segment
 53 *
 54 * @method
 55 * @param {sp.Vector2} v1 First vertex position
 56 * @param {sp.Vector2} v2 Second vertex position
 57 * @return {sp.Vector2} Normal of the segment
 58 */
 59 sp.Vector2.computeNormal = function (v1, v2) {
 60     var normal = new sp.Vector2(v1.y - v2.y, v2.x - v1.x);
 61     var length = Math.sqrt(normal.x * normal.x + normal.y * normal.y);
 62     if (length != 0) {
 63         normal.x = normal.x / length;
 64         normal.y = normal.y / length;
 65     }
 66     return normal;
 67 };

 69 /**
 70 * Compute the dot product of two vectors
 71 *
 72 * @method
 73 * @param {sp.Vector2} v1 First vector
 74 * @param {sp.Vector2} v2 Second vector
 75 * @return {float} Dot product
 76 */
 77 sp.Vector2.dotProduct = function (v1, v2) {
 78     return v1.x * v2.x + v1.y * v2.y;
 79 };
 80 
 81 /**
 82 * Addition of two vectors
 83 *
 84 * @method
 85 * @param {sp.Vector2} v1 First vector
 86 * @param {sp.Vector2} v2 Second vector
 87 * @return {sp.Vector2} Result vector
 88 */
 89 sp.Vector2.add = function (v1, v2) {
 90     return new sp.Vector2(v1.x + v2.x, v1.y + v2.y);
 91 };
 92 
 93 /**
 94 * Substraction of two vectors
 95 *
 96 * @method
 97 * @param {sp.Vector2} v1 First vector
 98 * @param {sp.Vector2} v2 Second vector
 99 * @return {sp.Vector2} Result vector
100 */
101 sp.Vector2.sub = function (v1, v2) {
102     return new sp.Vector2(v1.x - v2.x, v1.y - v2.y);
103 };
104 
105 /**
106 * Multiply a vector with a scalar
107 *
108 * @method
109 * @param {sp.Vector2} v Vector
110 * @param {float} s scalar
111 * @return {sp.Vector2} Result vector
112 */
113 sp.Vector2.mul = function (v, s) {
114     return new sp.Vector2(v.x * s, v.y * s);
115 };
116 
117 /**
118 * Divide a vector with a scalar
119 *
120 * @method
121 * @param {sp.Vector2} v Vector
122 * @param {float} s scalar
123 * @return {sp.Vector2} Result vector
124 */
125 sp.Vector2.div = function (v, s) {
126     return new sp.Vector2(v.x / s, v.y / s);
127 };
