 18 /**
 19 * Constructs Shape objects
 20 * @class Represents a Shape object
 21 * @extends sp.Transformable
 22 * @extends sp.Drawable
 23 * @param {int} pointCount New length of the point list
 24 * @param {sp.Texture} texture Texture
 25 */
 26 sp.Shape = function(pointCount, texture) {
 27     //Call base constructor
 28     sp.Transformable.call(this);
 29 
 30     /** @private */
 31     this.outlineThickness_ = 0;
 32     /** @private */
 33     this.outlineColor_ = new sp.Color();
 34 
 35     /** @private */
 36     this.vertexArray_ = new sp.VertexArray(sp.PrimitiveType.TRIANGLE_FAN, pointCount);
 37     /** @private */
 38     this.outlineVertexArray_ = new sp.VertexArray(sp.PrimitiveType.TRIANGLE_STRIP, (pointCount > 2) ? (pointCount * 2) + 2 : 0 );
 39 
 40     /** @private */
 41     this.needsUpdate_ = false;
 42     /** @private */
 43     this.needsColorUpdate_ = false;
 44     /** @private */
 45     this.needsTexCoordsUpdate_ = false;
 46 
 47     /** @private */
 48     this.texture_ = null;
 49     this.setTexture(texture);
 50 };
 51 sp.extend(sp.Shape, sp.Transformable);
 52 sp.extend(sp.Shape, sp.Drawable);
 53 
 54 /**
 55 * Resizes the point list, adding default points or removing existing ones to match the new length.
 56 *
 57 * @method
 58 * @param {int} pointCount New length of the point list
 59 */
 60 sp.Shape.prototype.setPointCount = function (pointCount) {
 61     this.vertexArray_.resize(pointCount);
 62     this.needsUpdate_ = true;
 63     this.needsColorUpdate_ = true;
 64     this.needsTexCoordsUpdate_ = true;
 65 };
 66 
 67 /**
 68 * Get the number of points
 69 *
 70 * @method
 71 * @returns {int} Length of point list
 72 */
 73 sp.Shape.prototype.getPointCount = function () {
 74     return this.vertexArray_.getVertexCount();
 75 };
 76 
 77 /**
 78 * Adds a new point to the shape
 79 *
 80 * @method
 81 * @param {sp.Vector2} position Position of the new point
 82 * @param {sp.Color} color Color of the new point
 83 */
 84 sp.Shape.prototype.addPoint = function (position, color) {
 85     this.vertexArray_.addVertex(sp.Vertex(position, color));
 86     this.needsUpdate_ = true;
 87     this.needsColorUpdate_ = true;
 88     this.needsTexCoordsUpdate_ = true;
 89 };
 90 
 91 /**
 92 * Sets the position of a point
 93 *
 94 * @method
 95 * @param {int} index Index of the point
 96 * @param {sp.Vector2} position New position of the point
 97 */
 98 sp.Shape.prototype.setPointPosition = function (index, position) {
 99     if (index < this.vertexArray_.getVertexCount()) {
100         var vertex = this.vertexArray_.getVertex(index);
101         vertex.position = position;
102         this.needsUpdate_ = true;
103         this.needsTexCoordsUpdate_ = true;
104     }
105 };
106 
107 /**
108 * Sets the color of a point
109 *
110 * @method
111 * @param {int} index Index of the point
112 * @param {sp.Color} position New color of the point
113 */
114 sp.Shape.prototype.setPointColor = function (index, color) {
115     if (index < this.vertexArray_.getVertexCount()) {
116         var vertex = this.vertexArray_.getVertex(index);
117         vertex.color = color;
118     }
119 };

121 /**
122 * Sets the outline thickness
123 *
124 * @method
125 * @param {float} outlineThickness Outline thickness
126 */
127 sp.Shape.prototype.setOutlineThickness = function (outlineThickness) {
128     this.outlineThickness_ = outlineThickness;
129     this.needsUpdate_ = true;
130 };
131 
132 /**
133 * Sets the outline color
134 *
135 * @method
136 * @param {sp.Color} color Color of the outline
137 */
138 sp.Shape.prototype.setOutlineColor = function (color) {
139     this.outlineColor_ = color;
140     this.needsColorUpdate_ = true;
141 };
142 
143 /**
144 * Updates the color of each vertex of the outline
145 *
146 * @method
147 */
148 sp.Shape.prototype.updateOutlineColor = function () {
149     var count = this.outlineVertexArray_.getVertexCount();
150 
151     for (var i = 0; i < count; i++) {
152         this.outlineVertexArray_.getVertex(i).color = this.outlineColor_;
153     }
154 };
155 
156 /**
157 * Updates the outline vertex array
158 *
159 * @method
160 */
161 sp.Shape.prototype.updateOutline = function () {
162     var count = this.vertexArray_.getVertexCount();
163     this.outlineVertexArray_.resize((count * 2) + 2);
164 
165     for (var i = 0; i < count; i++) {
166 
167         var v = this.vertexArray_.getVertex(i);
168         //Get two nearest vertices
169         var vLeft = (i == 0) ? this.vertexArray_.getVertex(count - 1) : this.vertexArray_.getVertex(i - 1);
170         var vRight = (i == count - 1) ? this.vertexArray_.getVertex(0) : this.vertexArray_.getVertex(i + 1);
171 
172         //Compute segment normals
173         var n1 = sp.Vector2.computeNormal(vLeft.position, v.position);
174         var n2 = sp.Vector2.computeNormal(v.position, vRight.position);
175 
176         //Normals must point towards outside of the shape
177         var tmp = sp.Vector2.sub(this.vertexArray_.getVertex(0).position, v.position);
178         if (sp.Vector2.dotProduct(n1, tmp) > 0) {
179             n1.x = -n1.x;
180             n1.y = -n1.y;
181         }
182         if (sp.Vector2.dotProduct(n2, tmp) > 0) {
183             n2.x = -n2.x;
184             n2.y = -n2.y;
185         }
186 
187         // Combine normals to get the extrusion direction
188         var factor = 1.0 + (n1.x * n2.x + n1.y * n2.y);
189         var normal = new sp.Vector2((n1.x + n2.x) / factor, (n1.y + n2.y) / factor);
190 
191         this.outlineVertexArray_.getVertex(i * 2).position = v.position;
192         this.outlineVertexArray_.getVertex(i * 2 + 1).position = new sp.Vector2(v.position.x + normal.x * this.outlineThickness_, v.position.y + normal.y * this.outlineThickness_);
193     }
194 
195     //The last point is the same as the first
196     this.outlineVertexArray_.getVertex(count * 2).position = this.outlineVertexArray_.getVertex(0).position;
197     this.outlineVertexArray_.getVertex(count * 2 + 1).position = this.outlineVertexArray_.getVertex(1).position;
198 };

200 /**
201 * Sets a texture to the shape.
202 *
203 * @method
204 * @param {sp.Texture} texture Texture
205 */
206 sp.Shape.prototype.setTexture = function (texture) {
207     this.texture_ = texture;
208 
209     if (this.texture_) {
210         this.needsTexCoordsUpdate_ = true;
211         var size = this.texture_.getSize();
212 
213         for (var i = 0; i < this.vertexArray_.getVertexCount(); i++) {
214             var v = this.vertexArray_.getVertex(i);
215             v.texCoords = new sp.Vector2(v.position.x / size.x, v.position.y / size.y);
216         }
217     }
218 
219 };
220 
221 /**
222 * Get the texture of the shape.
223 *
224 * @method
225 * @returns {sp.Texture} Texture
226 */
227 sp.Shape.prototype.getTexture = function () {
228     return this.texture_;
229 };
230 
231 /**
232 * Updates texture coordinates.
233 *
234 * @method
235 */
236 sp.Shape.prototype.updateTexCoords = function () {
237     if (this.texture_) {
238         var size = this.texture_.getSize();
239 
240         for (var i = 0; i < this.vertexArray_.getVertexCount(); i++) {
241             var v = this.vertexArray_.getVertex(i);
242             v.texCoords = new sp.Vector2(v.position.x / size.x, v.position.y / size.y);
243         }
244     }
245 };

247 /**
248 * Draws the shape in the given context.
249 *
250 * @method
251 * @param {sp.Context} context Context
252 / @param {sp.RenderOptions} renderOptions Optional render options
253 */
254 sp.Shape.prototype.draw = function (context, renderOptions) {
255     renderOptions.transform = this.getTransform();
256     renderOptions.texture = this.texture_;
257 
258     if (this.needsTexCoordsUpdate_) {
259         this.updateTexCoords();
260         this.needsTexCoordsUpdate_ = false;
261     }
262     
263     this.vertexArray_.draw(context, renderOptions);
264     if (this.outlineThickness_ > 0) {
265         renderOptions.texture = null;
266         //TODO: May be done on an update step, not when drawing
267         if (this.needsUpdate_) {
268             this.updateOutline();
269             this.needsUpdate_ = false;
270         }
271         if (this.needsColorUpdate_) {
272             this.updateOutlineColor();
273             this.needsColorUpdate_ = false;
274         }
275         this.outlineVertexArray_.draw(context, renderOptions);
276     }
277 };
